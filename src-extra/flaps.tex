\newcommand{\hsplit}[3]{%
  \begin{tikzpicture}
    \begin{scope}
      \node[inner sep=0pt,outer sep=0pt] (a) {\phantom{#1}};
      \clip (a.south east) rectangle ($(a.west)$);
      \node[inner sep=0pt,outer sep=0pt,text=#2]  {#1};
    \end{scope}
    \clip (a.north east) rectangle ($(a.west)$);
      \node[inner sep=0pt,outer sep=0pt,text=#3]  {#1};
  \end{tikzpicture}
}

\newcommand{\hsep}[1]{%
  \begin{mdframed}[userdefinedwidth=100pt,userdefinedheight=50pt]
    \begin{center}
      \hsplit{#1}{white}{black}      
    \end{center}
  \end{mdframed}
  \begin{mdframed}[userdefinedwidth=100pt]
    \begin{center}
      \hsplit{#1}{black}{white}      
    \end{center}
  \end{mdframed}
}

\end{multicols}
